pipeline{
    agent any
    tools{
        maven 'mvn'
    }
   
    stages{
        stage("SCM Checkout"){
            steps{
            git 'https://github.com/nakulshekar/employeeApp.git'
            }
        }
        stage("Maven Build"){
            steps{
                bat 'mvn install'
            }
        }
         stage("Docker Build"){
            steps{
                bat 'docker build -t nakulshekar/employee-app-spring .'
            }
        }
         stage("Push to Docker hub"){
            steps{
               script{
                 bat 'docker login -u nakulshekar -p coolspider'
                 bat 'docker push nakulshekar/employee-app-spring'
               }
                   }
       
    }
}
}
